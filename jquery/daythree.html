# 3.- Día 3

# Métodos de desvanecimiento de jQuery

        fadeIn() // Hace aparecer la imagen o objeto con difuminado
        fadeOut() // El objeto o etiqueta tiene que estar a la vista
        fadeToggle() // Hace 2 opciones aparecer y desaparecer objeto
        fadeTo() // Hace desaparecer la imagen o objeto con difuminado pero no se borra completamente

# fadeIn()

Mientras no cambie las posiciones a los elementos se puede utilizar 'fadeIn'

      $('etiqueta & clase $ id').fadeIn('aca se pone los segundos')

# Métodos deslizantes de jQuery

        slideDown()  // El objeto tiene que estar oculto para que funcione
        slideUp()   // El objeto tiene que estar  a la vista y el metodo lo esconde
        slideToggle()  // Realiza las 2 funciones ocultar y mostrar

# Funcion Hover (mouseover, mouseout)

        mouseover()  // Muestra el objeto
        mouseout()   // Esconde el objeto
