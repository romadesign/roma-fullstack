# Para iniciar primero debemos de realizar algunas instalaciones necesarias

Antes vamos a realizar unas configuración de instalación

# Instalar Node y NPM con apt desde los respositorios

        $ sudo apt update
        $ sudo apt install nodejs
        $ node -v
        $ sudo apt install npm
        $ npm -v

# Instalar Node y NPM con apt usando un repositorio PPA de NodeSource

        $ curl -sL https://deb.nodesource.com/setup_14.x -o nodesource_setup.sh
        $ sudo bash nodesource_setup.sh
        $ sudo apt install nodejs
        $ node -v
        $ npm -v

# Instalar Visual Studio Code en Debian, Ubuntu y derivados

        $ curl https://packages.microsoft.com/keys/microsoft.asc | gpg --dearmor > microsoft.gpg && \
        $ sudo mv microsoft.gpg /etc/apt/trusted.gpg.d/microsoft.gpg && \
        $ sudo sh -c 'echo "deb [arch=amd64] https://packages.microsoft.com/repos/vscode stable main" > /etc/apt/sources.list.d/vscode.list' && \
        $ sudo apt-get update && \
        $ sudo apt install code code-insiders
